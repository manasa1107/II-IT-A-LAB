import java.util.*;
public class Main
{
    public static void main(String[] args)
    {
        Scanner s = new Scanner(System.in);
        System.out.print("Enter string : ");
        String str = s.next();
        int n = str.length();
        String rev = "";
        for(int i=n-1; i>=0; i--)
        {
            rev += str.charAt(i);
        }
        System.out.println("Reverse of string -> " + rev);
    }
}
