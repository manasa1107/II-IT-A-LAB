import java.util.*;
public class Main
{
    public static void main(String[] args)
    {
        Scanner s = new Scanner(System.in);
        System.out.print("Enter the number : ");
        int n = s.nextInt();
        System.out.print("Enter the power : ");
        int m = s.nextInt();
        int pow=1;
        for(int i=0; i<m; i++)
        {
            pow *= n;
        }
        System.out.println("Power(" + n + " , " + m + ") = " + pow);
    }
}
