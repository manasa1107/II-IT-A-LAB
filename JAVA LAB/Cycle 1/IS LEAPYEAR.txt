import java.util.*;
public class Main
{
    public static void main(String[] args)
    {
        Scanner s = new Scanner(System.in);
        System.out.print("Enter the year : ");
        int n = s.nextInt();
        if(((n % 4 == 0) && (n % 100 != 0)) || (n % 400 == 0))
        {
            System.out.println(n + " is LEAP YEAR");
        }
        else
        {
            System.out.println(n + " is NOT a LEAP YEAR");
        }
    }
}
