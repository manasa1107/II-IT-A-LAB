public class Main
{
    public static void main(String[] args)
    {
        char ch;
        String s;
        ch = 'A';
        s = Character.toString(ch);
        System.out.println("Char to String : " + s);
        String str = "Java";
        int n = str.length();
        for(int i=0; i<n; i++)
        {
            System.out.println("char at position " + i +" is -> " + str.charAt(i));
        }
    }
}
